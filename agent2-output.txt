=== DEPLOY_SCRIPTS (exit 0) ===
=== DEPLOY.SH ===
#!/bin/bash
# PS VIBE Sales Bot Deploy Script (Main VPS)
# Usage: ./deploy.sh [source_dir] [target_dir]

set -e
SOURCE="${1:-/root/staging/bot_src}"
TARGET="${2:-/root/psvibe-sales-bot}"
BACKUP="/root/backups/predeploy_$(date +%Y%m%d_%H%M%S).tar.gz"
LOG="/root/staging/scripts/deploy.log"

echo "[$(date)] Starting deploy: $SOURCE → $TARGET" | tee -a "$LOG"

# Backup current
if [ -d "$TARGET" ]; then
    echo "[$(date)] Backing up current to $BACKUP" | tee -a "$LOG"
    mkdir -p /root/backups
    tar czf "$BACKUP" -C "$(dirname $TARGET)" "$(basename $TARGET)" 2>/dev/null
fi

# Stop service
echo "[$(date)] Stopping psvibe-sale-bot service..." | tee -a "$LOG"
systemctl stop psvibe-sale-bot 2>/dev/null || true
sleep 2

# Copy new code
if [ -d "$SOURCE" ]; then
    echo "[$(date)] Copying new code..." | tee -a "$LOG"
    rm -rf "$TARGET"
    cp -a "$SOURCE" "$TARGET"
else
    echo "[$(date)] ERROR: Source $SOURCE not found!" | tee -a "$LOG"
    exit 1
fi

# Install deps
if [ -f "$TARGET/requirements.txt" ]; then
    echo "[$(date)] Installing Python dependencies..." | tee -a "$LOG"
    /root/venv/bin/pip install -r "$TARGET/requirements.txt" 2>&1 | tail -5 >> "$LOG"
fi

# Start service
echo "[$(date)] Starting psvibe-sale-bot service..." | tee -a "$LOG"
systemctl daemon-reload
systemctl start psvibe-sale-bot
sleep 3

# Verify
PID=$(systemctl show psvibe-sale-bot -p MainPID 2>/dev/null | cut -d= -f2)
if [ "$PID" -gt 0 ] 2>/dev/null; then
    echo "[$(date)] ✅ Deploy SUCCESSFUL! PID: $PID" | tee -a "$LOG"
else
    echo "[$(date)] ❌ Deploy FAILED! Rolling back..." | tee -a "$LOG"
    /root/staging/scripts/rollback.sh "$BACKUP" "$TARGET"
    exit 1
fi
=== ROLLBACK.SH ===
#!/bin/bash
# Rollback deploy
BACKUP="$1"
TARGET="${2:-/root/psvibe-sales-bot}"

if [ -f "$BACKUP" ]; then
    echo "[$(date)] Rolling back from $BACKUP"
    systemctl stop psvibe-sale-bot 2>/dev/null || true
    sleep 2
    rm -rf "$TARGET"
    tar xzf "$BACKUP" -C "$(dirname $TARGET)"
    systemctl daemon-reload
    systemctl start psvibe-sale-bot
    echo "Rollback complete"
else
    echo "No backup found at $BACKUP"
    exit 1
fi
